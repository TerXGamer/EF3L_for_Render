  // تفعيل الدخول السري عبر لوحة المفاتيح: اضغط على (Ctrl + Shift + A) معاً لفتح اللوحة
  document.addEventListener('keydown', function(e) {
    if (e.ctrlKey && e.shiftKey && e.key.toLowerCase() === 'a') {
      e.preventDefault();
      const overlay = document.getElementById('adminOverlay');
      overlay.style.display = overlay.style.display === 'none' ? 'block' : 'none';
      
      // تعبئة الرمز السري تلقائياً من الذاكرة المحلية إذا تم إدخاله بنجاح سابقاً
      const savedSecret = localStorage.getItem('ifal_admin_secret');
      if (savedSecret) {
        document.getElementById('adminSecretInput').value = savedSecret;
      }
    }
  });

  // جلب قائمة المستخدمين من السيرفر
  async function loadAdminUsers() {
    const secret = document.getElementById('adminSecretInput').value.trim();
    if (!secret) return alert('برجاء إدخال رمز الأمان أولاً للتحقق من هويتك.');

    try {
      const response = await fetch('/api/admin/users', {
        headers: { 'X-Admin-Secret': secret }
      });
      
      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.error || 'فشل الاتصال بالخادم السري.');
      }

      const data = await response.json();
      localStorage.setItem('ifal_admin_secret', secret); // حفظ المفتاح محلياً لراحتك في المرات القادمة
      
      document.getElementById('adminContentSection').style.display = 'block';
      document.getElementById('userCount').innerText = data.users.length;
      
      const tbody = document.getElementById('adminUsersTableBody');
      tbody.innerHTML = '';
      
      data.users.forEach(user => {
        const tr = document.createElement('tr');
        tr.style.borderBottom = '1px solid var(--line)';
        tr.innerHTML = `
          ${user.username}
          ${user.name || 'لا يوجد'}
          ${user.email || 'لا يوجد'}
          
            
          
        `;
        tbody.appendChild(tr);
      });
    } catch (err) {
      alert(err.message);
    }
  }

  // جلب بيانات مستخدم محدد عند الضغط (On-Demand Inspect)
  async function inspectUserData(username) {
    const secret = document.getElementById('adminSecretInput').value.trim();
    try {
      const response = await fetch(`/api/admin/user-data?username=${encodeURIComponent(username)}`, {
        headers: { 'X-Admin-Secret': secret }
      });
      
      if (!response.ok) {
        const err = await response.json();
        throw new Error(err.error || 'عذراً، تعذر جلب كشف البيانات لهذا المستخدم.');
      }

      const data = await response.json();
      document.getElementById('userDataInspectSection').style.display = 'block';
      document.getElementById('inspectingUsername').innerText = username;
      
      // عرض بيانات المستخدم المخزنة بملف الـ JSON بشكل منظم جداً ومقروء
      document.getElementById('userDataJsonContainer').innerText = JSON.stringify(data.userData, null, 2);
    } catch (err) {
      alert(err.message);
    }
  }
